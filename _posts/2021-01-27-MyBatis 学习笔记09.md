<p align="center">一对一查询方式一</p>

---

```
数据库表
CREATE TABLE team (teamid INT, teamname VARCHAR(20), cid INT);
CREATE TABLE city (cityid INT, cityname VARCHAR(20));
```

```
与数据库表对应的类
public class City implements Serializable {

    private int cityid;
    private String cityname;
    省略 Get、Set 方法
}
```

```
与数据库表对应的类
public class Team implements Serializable {

    private int cid;
    private int teamid;
    private String teamname;
    省略 Get、Set 方法
}
```

```
与查询结果对应的实体类，也可以通过继承 City 或 Team 实现
public class CityTeam implements Serializable {

    private int teamid;
    private int cityid;
    private String cityname;
    private String teamname;
    省略 Get、Set 方法
}
```

```
<!-- 持久层的映射 -->
<mapper namespace="chu.yi.bo.dao.CityTeamDao">
    <select id="findAll" resultType="chu.yi.bo.domain.CityTeam">
        select t.teamid,t.teamname,c.cityid,c.cityname from team t,city c where t.cid = c.cityid;
    </select>
</mapper>
```

<p align="center">一对一查询方式二</p>

----

```
与数据库表对应的类
public class City implements Serializable {

    private int cityid;
    private String cityname;
    private Team team;
    省略 Get、Set 方法
}
```

```
与数据库表对应的类
public class Team implements Serializable {

    private int cid;
    private int teamid;
    private String teamname;
    省略 Get、Set 方法    
}
```

```
<!-- 持久层的映射 -->
<mapper namespace="chu.yi.bo.dao.CityDao">
    <resultMap id="cityMap" type="chu.yi.bo.domain.City">
        <id column="cityid" property="cityid"/>
        <result column="cityname" property="cityname"/>
        <association property="team" javaType="chu.yi.bo.domain.Team">
            <id column="teamid" property="teamid"/>
            <result column="teamname" property="teamname"/>
            <result column="cid" property="cid"/>
        </association>
    </resultMap>

    <select id="findAll" resultMap="husbandMap">
        select t.*,c.* from team t,City c where t.cid = c.cityid;
    </select>
</mapper>
```

<p align="center">一对多查询</p>

----

```
CREATE TABLE user (id INT, username VARCHAR(20));
CREATE TABLE orderinfo (orderid INT, uid INT, goods VARCHAR(20));
```

```
public class OrderInfo implements Serializable {

    private int orderid;
    private int uid;
    private String goods;
    省略 Get、Set 方法
}
```

```
public class User implements Serializable {

    private int id;
    private String username;
    private List<OrderInfo> orderList;
    省略 Get、Set 方法
}
```

```
<resultMap id="userMap" type="chu.yi.bo.domain.User">
    <id column="id" property="id"/>
    <result column="username" property="username"/>
    <collection property="orderInfoList" ofType="chu.yi.bo.domain.OrderInfo">
        <id column="orderid" property="orderid"/>
        <result column="uid" property="uid"/>
        <result column="goods" property="goods"/>
    </collection>
</resultMap>

<!-- 配置查询所有操作 -->
<select id="findAll" resultMap="userMap">
    select u.*,o.* from user u left outer join orderinfo o on u.id = o.uid;
</select>
```

<p align="center">多对多查询</p>

----

```
CREATE TABLE teacher (teacherId INT, username VARCHAR(20), age INT, sex VARCHAR(1));
CREATE TABLE student (id INT, username VARCHAR(20), age INT, sex VARCHAR(1));
CREATE TABLE mapper (tid INT, sid INT);
```

```
public class Teacher implements Serializable {

    private int teacherId;
    private String username;
    private int age;
    private String sex;
    private List<Student> studentList;
    省略 Get、Set 方法
}
```

```
public class Student implements Serializable {

    private int studentId;
    private String username;
    private int age;
    private String sex;
    省略 Get、Set 方法
}
```

```
<resultMap id="teacherMap" type="chu.yi.bo.domain.Teacher">
    <id column="teacherId" property="teacherId"/>
    <result column="username" property="username"/>
    <result column="age" property="age"/>
    <result column="sex" property="sex"/>
    <collection property="studentList" ofType="chu.yi.bo.domain.Student">
        <id column="studentId" property="studentId"/>
        <result column="username" property="username"/>
        <result column="age" property="age"/>
        <result column="sex" property="sex"/>
    </collection>
</resultMap>

<select id="findAll" resultMap="teacherMap">
    SELECT t.*,s.* FROM teacher t INNER JOIN mapper m ON t.teacherId = m.tid INNER JOIN student s ON s.studentId = m.sid;
</select>
```
